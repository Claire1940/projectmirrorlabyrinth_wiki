  <Step num="1" title="Periksa Status Komunitas Resmi">

  </Step>
  <Step num="2" title="Tukarkan Kode Baru">

  <Step num="3" title="Mulai Luxcavation">

  </Step>
  <Step num="4" title="Kelola Refleksi">

  <Step num="5" title="Defragmentasi dan Upgrade">
